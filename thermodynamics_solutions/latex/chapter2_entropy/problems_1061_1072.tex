\section{Problem 1061}
\textbf{(Various Sources)}

Additional entropy problems covering mixing entropy, irreversible processes, and the Clausius inequality.

\subsection*{Key Formulas}

\textbf{Clausius Inequality:}
\begin{equation}
\oint \frac{dQ}{T} \leq 0
\end{equation}

\textbf{Entropy of Mixing (Ideal Gases):}
\begin{equation}
\Delta S_{mix} = -R\sum_i n_i \ln x_i
\end{equation}

\textbf{General Entropy Change:}
\begin{equation}
dS = \frac{dU + pdV}{T}
\end{equation}
